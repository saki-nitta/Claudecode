/**
 * おもちゃのバトン - プライスカード自動生成スクリプト（v2）
 * 
 * 【使い方】
 * 1. Googleスプレッドシートを開く
 * 2. 拡張機能 → Apps Script → 既存コードを全削除してこれを貼り付け
 * 3. 保存後、スプレッドシートをリロード
 * 4. メニュー「🧸 おもちゃのバトン」→「プライスカードを生成」を実行
 * 5. 「プライスカード」シートが自動生成されます（件数に応じて自動で枚数調整）
 */

function generatePriceCards() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  
  // ─── 仕入れ台帳からデータ取得 ───
  const sourceSheet = ss.getSheetByName('仕入れ台帳');
  if (!sourceSheet) {
    SpreadsheetApp.getUi().alert('「仕入れ台帳」シートが見つかりません。');
    return;
  }
  
  const lastRow = sourceSheet.getLastRow();
  if (lastRow < 2) {
    SpreadsheetApp.getUi().alert('仕入れ台帳にデータがありません。');
    return;
  }
  
  // A〜S列を取得（1行目はヘッダー）
  const allData = sourceSheet.getRange(2, 1, lastRow - 1, 19).getValues();
  
  // D列（index 3）が「販売中」のものだけ抽出
  const salesItems = allData.filter(row => row[3] === '販売中');
  
  // ─── プライスカードシートの準備 ───
  let cardSheet = ss.getSheetByName('プライスカード');
  if (cardSheet) {
    ss.deleteSheet(cardSheet);
  }
  cardSheet = ss.insertSheet('プライスカード');
  
  const COLS_PER_CARD = 9;   // カード1枚分の列数
  const ROWS_PER_CARD = 10;  // カード1枚分の行数
  const MARGIN_COL = 1;      // カード間の余白列
  const MARGIN_ROW = 1;      // カード間の余白行
  const START_COL = 2;       // 開始列（B列）
  const START_ROW = 2;       // 開始行
  
  // 件数に応じてカード総数・行数を自動算出（最低10枚＝1ページ分は確保）
  const totalCards = Math.max(10, salesItems.length);
  const totalCardRows = Math.ceil(totalCards / 2);
  
  // ─── シート全体の設定 ───
  cardSheet.setColumnWidths(1, 20, 28);
  cardSheet.setColumnWidth(1, 20);
  
  for (let col = 0; col < 2; col++) {
    const baseCol = START_COL + col * (COLS_PER_CARD + MARGIN_COL);
    for (let c = 0; c < COLS_PER_CARD; c++) {
      cardSheet.setColumnWidth(baseCol + c, 30);
    }
    if (col < 1) cardSheet.setColumnWidth(baseCol + COLS_PER_CARD, 12);
  }
  
  for (let row = 0; row < totalCardRows; row++) {
    const baseRow = START_ROW + row * (ROWS_PER_CARD + MARGIN_ROW);
    for (let r = 0; r < ROWS_PER_CARD; r++) {
      cardSheet.setRowHeight(baseRow + r, 17);
    }
    if (row < totalCardRows - 1) cardSheet.setRowHeight(baseRow + ROWS_PER_CARD, 8);
  }
  
  // ─── カードを描画（totalCards枚分） ───
  for (let i = 0; i < totalCards; i++) {
    const col = i % 2;
    const row = Math.floor(i / 2);
    
    const baseCol = START_COL + col * (COLS_PER_CARD + MARGIN_COL);
    const baseRow = START_ROW + row * (ROWS_PER_CARD + MARGIN_ROW);
    
    if (i < salesItems.length) {
      drawCard(cardSheet, baseRow, baseCol, ROWS_PER_CARD, COLS_PER_CARD, salesItems[i], true);
    } else {
      drawCard(cardSheet, baseRow, baseCol, ROWS_PER_CARD, COLS_PER_CARD, null, false);
    }
  }
  
  // ─── 印刷設定 ───
  const sheet = cardSheet;
  sheet.setHiddenGridlines(true);
  
  const printRange = sheet.getRange(
    1, 1,
    START_ROW + totalCardRows * (ROWS_PER_CARD + MARGIN_ROW),
    START_COL + 2 * (COLS_PER_CARD + MARGIN_COL)
  );
  
  SpreadsheetApp.getUi().alert(
    '✅ プライスカードを生成しました！\n\n' +
    '販売中アイテム：' + salesItems.length + '件\n' +
    '白無地テンプレート：' + (totalCards - salesItems.length) + '枚\n' +
    '生成枚数：' + totalCards + '枚（' + totalCardRows + '行×2列）\n\n' +
    '印刷は ファイル → 印刷 → 用紙サイズ「A4」縦 でご確認ください。\n' +
    '※10枚を超える分は自動で複数ページに分かれます。'
  );
}

/**
 * 1枚のカードを描画する
 */
function drawCard(sheet, baseRow, baseCol, rows, cols, item, hasData) {
  
  const COLOR_BG       = '#fdf8f3';
  const COLOR_BORDER   = '#d0bfa7';
  const COLOR_BRAND    = '#7a5c3e';
  const COLOR_ITEM     = '#3d2b1f';
  const COLOR_PRICE    = '#c85a32';
  const COLOR_LISTPRICE= '#9e9e9e';
  const COLOR_NUM      = '#b0a090';
  
  const cardRange = sheet.getRange(baseRow, baseCol, rows, cols);
  
  cardRange.setBackground(hasData ? COLOR_BG : '#ffffff');
  
  cardRange.setBorder(
    true, true, true, true, false, false,
    COLOR_BORDER,
    SpreadsheetApp.BorderStyle.DASHED
  );
  
  if (!hasData) {
    const emptyCell = sheet.getRange(baseRow + Math.floor(rows/2), baseCol, 1, cols);
    emptyCell.setValue('　✏️ 手書き用');
    emptyCell.setFontColor('#d0bfa7')
             .setFontSize(8)
             .setHorizontalAlignment('center')
             .setVerticalAlignment('middle');
    return;
  }
  
  // 管理番号（左上）
  const numCell = sheet.getRange(baseRow, baseCol, 1, 3);
  numCell.mergeAcross();
  numCell.setValue(item[0] || '');
  numCell.setFontColor(COLOR_NUM)
         .setFontSize(6)
         .setHorizontalAlignment('left')
         .setVerticalAlignment('middle')
         .setFontFamily('Meiryo UI');
  sheet.getRange(baseRow, baseCol).setNumberFormat('@');
  
  // ブランド名
  const brandCell = sheet.getRange(baseRow + 1, baseCol, 2, cols);
  brandCell.merge();
  brandCell.setValue(item[4] || '');
  brandCell.setFontColor(COLOR_BRAND)
           .setFontSize(8)
           .setFontWeight('bold')
           .setHorizontalAlignment('left')
           .setVerticalAlignment('bottom')
           .setFontFamily('Meiryo UI');
  
  // 区切り線
  const divider = sheet.getRange(baseRow + 3, baseCol, 1, cols);
  divider.mergeAcross();
  divider.setBorder(false, false, true, false, false, false, COLOR_BORDER, SpreadsheetApp.BorderStyle.SOLID);
  
  // 品名
  const nameCell = sheet.getRange(baseRow + 3, baseCol, 3, cols);
  nameCell.merge();
  nameCell.setValue(item[2] || '');
  nameCell.setFontColor(COLOR_ITEM)
          .setFontSize(11)
          .setFontWeight('bold')
          .setHorizontalAlignment('left')
          .setVerticalAlignment('middle')
          .setWrap(true)
          .setFontFamily('Meiryo UI');
  
  // 定価（取り消し線）
  const listPriceCell = sheet.getRange(baseRow + 6, baseCol, 1, cols);
  listPriceCell.merge();
  const listPrice = item[18];
  const listPriceText = listPrice ? '定価 ¥' + Number(listPrice).toLocaleString() : '';
  listPriceCell.setValue(listPriceText);
  listPriceCell.setFontColor(COLOR_LISTPRICE)
               .setFontSize(7)
               .setHorizontalAlignment('right')
               .setVerticalAlignment('middle')
               .setFontFamily('Meiryo UI');
  if (listPriceText) {
    listPriceCell.setFontLine('line-through');
  }
  
  // 販売価格（大きく・赤茶）
  const priceCell = sheet.getRange(baseRow + 7, baseCol, 3, cols);
  priceCell.merge();
  const salePrice = item[16];
  const salePriceText = salePrice ? '¥' + Number(salePrice).toLocaleString() + '（税込）' : '';
  priceCell.setValue(salePriceText);
  priceCell.setFontColor(COLOR_PRICE)
           .setFontSize(13)
           .setFontWeight('bold')
           .setHorizontalAlignment('center')
           .setVerticalAlignment('middle')
           .setFontFamily('Meiryo UI');
}

/**
 * スプレッドシートにメニューを追加
 */
function onOpen() {
  SpreadsheetApp.getUi()
    .createMenu('🧸 おもちゃのバトン')
    .addItem('プライスカードを生成', 'generatePriceCards')
    .addToUi();
}
